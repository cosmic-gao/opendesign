## 使用示例

### 基础用法（浅合并）

```typescript
import { merge, DEFAULT_BREAKPOINTS } from '@openlayout/config';
import type { CSSProperties as ReactCSSProperties } from 'react';

// 合并默认配置（浅合并）
const config = merge<React.ReactElement, React.CSSProperties>({
  header: { fixed: true },
  sidebar: { collapsed: true }
});

// 结果：LayoutConfig（所有字段都是必需的）
// {
//   header: { enabled: true, height: 64, fixed: true, full: false },
//   footer: { enabled: true, height: 48, fixed: false, full: false },
//   sidebar: { enabled: true, width: 200, min: 80, collapsible: false, collapsed: true, full: true, overlay: false },
//   content: { enabled: true, scrollable: true },
//   animation: { enabled: true, duration: 200, easing: 'ease' },
//   breakpoints: { xs: 480, sm: 576, md: 768, lg: 992, xl: 1200, xxl: 1400 },
//   mobileBreakpoint: 768
// }
```

### 深度合并

```typescript
// 深度合并：部分覆盖断点时保留其他断点
const config = merge<React.ReactElement, React.CSSProperties>(
  {
    breakpoints: { lg: 1000 }
  },
  { deep: true }
);

// 结果：其他断点保留，只有 lg 被覆盖
// { xs: 480, sm: 576, md: 768, lg: 1000, xl: 1200, xxl: 1400 }
```

### 自定义断点

```typescript
const config = merge<React.ReactElement, React.CSSProperties>({
  breakpoints: {
    xs: 320,
    sm: 480,
    md: 640,
    lg: 1024,
    xl: 1440,
    xxl: 1920
  },
  mobileBreakpoint: 640
});
```

### 禁用组件

```typescript
const config = merge<React.ReactElement, React.CSSProperties>({
  header: { enabled: false },
  sidebar: { enabled: false }
});
```

### 配置验证

```typescript
import { validate } from '@openlayout/config';

const result = validate<React.ReactElement, React.CSSProperties>({
  header: { height: 200 },
  sidebar: { width: 50, min: 100 },  // 错误：min > width
  animation: { duration: -10 }         // 错误：duration < 0
});

if (!result.valid) {
  console.log(result.errors);
  // [
  //   { field: 'sidebar.min', message: 'sidebar.min must be <= sidebar.width', value: 100 },
  //   { field: 'animation.duration', message: 'animation.duration must be >= 0', value: -10 }
  // ]
}
```

### MergeOptions 接口

  deep?: boolean;  // 是否深度合并，默认 false
}

## 验证函数

### ValidationError 接口

```typescript
export interface ValidationError {
  field: string;      // 字段路径，如 "sidebar.width"
  message: string;    // 错误描述
  value?: unknown;    // 实际值
}
```

### ValidationResult 接口

```typescript
export interface ValidationResult {
  valid: boolean;
  errors: ValidationError[];
}
```

### validate 函数（泛型）

```typescript
export function validate<VNode, CSSProperties>(
  props: LayoutProps<VNode, CSSProperties>
): ValidationResult
```

**验证规则**:

| 字段 | 规则 | 错误信息 |
|------|------|----------|
| `header.height` | 1-10000 | "header.height must be between 1 and 10000" |
| `footer.height` | 1-10000 | "footer.height must be between 1 and 10000" |
| `sidebar.width` | 1-10000 | "sidebar.width must be between 1 and 10000" |
| `sidebar.min` | 1-10000 | "sidebar.min must be between 1 and 10000" |
| `sidebar.min` | <= width | "sidebar.min must be <= sidebar.width" |
| `animation.duration` | >= 0 | "animation.duration must be >= 0" |
| `animation.easing` | 关键字或CSS函数 | "animation.easing must be one of: ease, linear, ease-in, ..." |
| `breakpoints.{key}` | > 0 | "breakpoints.{key} must be a number > 0" |
| `breakpoints` | 升序排列 | "breakpoints values must be in ascending order" |
| `mobileBreakpoint` | 匹配断点值 | "mobileBreakpoint must match one of breakpoints values" |
