## 2. Deployment 配置

```yaml
# graph-runtime-deployment.yaml
apiVersion: apps/v1
kind: Deployment
metadata:
  name: graph-runtime
  namespace: graph-system
spec:
  replicas: 3
  selector:
    matchLabels:
      app: graph-runtime
  template:
    metadata:
      labels:
        app: graph-runtime
    spec:
      containers:
      - name: graph-runtime
        image: graph-runtime:latest
        ports:
        - containerPort: 8080
        env:
        - name: NODE_ENV
          value: production
        - name: REDIS_HOST
          value: redis-cluster
        - name: KAFKA_BROKERS
          value: kafka-cluster:9092
        resources:
          requests:
            memory: "256Mi"
            cpu: "250m"
          limits:
            memory: "2Gi"
            cpu: "2000m"
        livenessProbe:
          httpGet:
            path: /health
            port: 8080
          initialDelaySeconds: 30
          periodSeconds: 10
        readinessProbe:
          httpGet:
            path: /ready
            port: 8080
          initialDelaySeconds: 5
          periodSeconds: 5
```

## 3. Service 配置

```yaml
# graph-runtime-service.yaml
apiVersion: v1
kind: Service
metadata:
  name: graph-runtime
  namespace: graph-system
spec:
  selector:
    app: graph-runtime
  ports:
  - name: http
    port: 8080
    targetPort: 8080
  - name: grpc
    port: 9090
    targetPort: 9090
  # Headless Service for StatefulSet
  clusterIP: None
```

## 4. HPA 配置

```yaml
# graph-runtime-hpa.yaml
apiVersion: autoscaling/v2
kind: HorizontalPodAutoscaler
metadata:
  name: graph-runtime-hpa
  namespace: graph-system
spec:
  scaleTargetRef:
    apiVersion: apps/v1
    kind: Deployment
    name: graph-runtime
  minReplicas: 3
  maxReplicas: 20
  metrics:
  - type: Resource
    resource:
      name: cpu
      target:
        type: Utilization
        averageUtilization: 70
  - type: Pods
    pods:
      metric:
        name: graph_running_graphs
      target:
        type: AverageValue
        averageValue: "10"
  behavior:
    scaleUp:
      stabilizationWindowSeconds: 60
      policies:
      - type: Percent
        value: 100
        periodSeconds: 15
    scaleDown:
      stabilizationWindowSeconds: 300
      policies:
      - type: Percent
        value: 10
        periodSeconds: 60
```

  name: graph-runtime-config
  namespace: graph-system

  config.yaml: |
    graph:
      maxSupersteps: 10000
      checkpointFrequency: 5
      defaultTimeout: 30000
    
    router:
      backend: kafka
      kafka:
        topicPrefix: graph
        consumerGroup: graph-runtime
    
    checkpoint:
      storage: s3
      s3:
        bucket: graph-checkpoints
        region: us-east-1
    
    logging:

## 6. StatefulSet (可选，用于需要持久化的场景)

```yaml
# graph-runtime-statefulset.yaml
apiVersion: apps/v1
kind: StatefulSet
metadata:
  name: graph-runtime
  namespace: graph-system
spec:
  serviceName: graph-runtime
  replicas: 3
  selector:
    matchLabels:
      app: graph-runtime
  template:
    metadata:
      labels:
        app: graph-runtime
    spec:
      containers:
      - name: graph-runtime
        image: graph-runtime:latest
        volumeMounts:
        - name: data
          mountPath: /data
  volumeClaimTemplates:
  - metadata:
      name: data
    spec:
      accessModes: [ "ReadWriteOnce" ]
      resources:
        requests:
          storage: 10Gi
```

## 7. 网络策略

```yaml
# network-policy.yaml
apiVersion: networking.k8s.io/v1
kind: NetworkPolicy
metadata:
  name: graph-runtime-network-policy
  namespace: graph-system
spec:
  podSelector:
    matchLabels:
      app: graph-runtime
  policyTypes:
  - Ingress
  - Egress
  ingress:
  - from:
    - podSelector:
        matchLabels:
          app: ingress-nginx
    ports:
    - protocol: TCP
      port: 8080
  egress:
  - to:
    - podSelector:
        matchLabels:
          app: redis
    ports:
    - protocol: TCP
      port: 6379
  - to:
    - podSelector:
        matchLabels:
          app: kafka
    ports:
    - protocol: TCP
      port: 9092
```
